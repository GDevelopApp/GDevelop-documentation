44  const gd::String& GetOutput() { return output; };
45 
46  protected:
47  void OnVisitSubExpressionNode(SubExpressionNode& node) override {
48  output += "(";
49  node.expression->Visit(*this);
50  output += ")";
51  }
52  void OnVisitOperatorNode(OperatorNode& node) override {
53  node.leftHandSide->Visit(*this);
54  if (node.op == ' ') {
55  // There is no "space" operator. If it's there, it's because
56  // an operator could not be found (that's an error). Add only
57  // a whitespace between terms.
58  output += " ";
59  } else {
60  output += " ";
61  output.push_back(node.op);
62  output += " ";
63  }
64  node.rightHandSide->Visit(*this);
65  }
66  void OnVisitUnaryOperatorNode(UnaryOperatorNode& node) override {
67  output.push_back(node.op);
68  node.factor->Visit(*this);
69  }
70  void OnVisitNumberNode(NumberNode& node) override { output += node.number; }
71  void OnVisitTextNode(TextNode& node) override {
72  output +=
73  "\"" +
74  node.text.FindAndReplace("\\", "\\\\").FindAndReplace("\"", "\\\"") +
75  "\"";
76  }
77  void OnVisitVariableNode(VariableNode& node) override {
78  output += node.name;
79  if (node.child) node.child->Visit(*this);
80  }
81  void OnVisitVariableAccessorNode(VariableAccessorNode& node) override {
82  output += "." + node.name;
83  if (node.child) node.child->Visit(*this);
84  }
85  void OnVisitVariableBracketAccessorNode(
86  VariableBracketAccessorNode& node) override {
87  output += "[";
88  node.expression->Visit(*this);
89  output += "]";
90  if (node.child) node.child->Visit(*this);
91  }
92  void OnVisitIdentifierNode(IdentifierNode& node) override {
93  output += node.identifierName;
94  if (!node.childIdentifierName.empty()) {
95  output += "." + node.childIdentifierName;
96  }
97  }
98  void OnVisitObjectFunctionNameNode(ObjectFunctionNameNode& node) override {
99  if (!node.behaviorFunctionName.empty()) {
100  output +=
101  node.objectName + "." + node.objectFunctionOrBehaviorName + "::" + node.behaviorFunctionName;
102  } else {
103  output += node.objectName + "." + node.objectFunctionOrBehaviorName;
104  }
105  };
106  void OnVisitFunctionCallNode(FunctionCallNode& node) override {
107  if (!node.behaviorName.empty()) {
108  output +=
109  node.objectName + "." + node.behaviorName + "::" + node.functionName;
110  } else if (!node.objectName.empty()) {
111  output += node.objectName + "." + node.functionName;
112  } else {
113  output += node.functionName;
114  }
115 
116  output += "(";
117 
118  bool isFirst = true;
119  for (auto& parameterNode : node.parameters) {
120  if (!isFirst) output += ", ";
121  parameterNode->Visit(*this);
122  isFirst = false;
123  }
124 
125  output += ")";
126  }
127  void OnVisitEmptyNode(EmptyNode& node) override { output += node.text; }
128 
129  private:
130  gd::String output;
131 };
