45 
49  size_t GetStartPosition() const { return startPosition; }

57 
58  gd::String ToString() const {
59  return (colorationKind == 0 ? "String"
60  : colorationKind == 1 ? "Number"
61  : colorationKind == 2 ? "Object"
62  : colorationKind == 3 ? "Variable"
63  : colorationKind == 4 ? "Operator"
64  : "Unknown") +
65  gd::String(" [") + gd::String::From(startPosition) + " " +
66  gd::String::From(endPosition) + "[";
67  }
68 
70  ExpressionColorationDescription() : colorationKind(String){};
71 
72  ExpressionColorationDescription(ColorationKind colorationKind_,
73  size_t startPosition_, size_t endPosition_)
74  : colorationKind(colorationKind_), startPosition(startPosition_),
75  endPosition(endPosition_) {}
76 
77 private:
78  ColorationKind colorationKind;
79  size_t startPosition = 0;
80  size_t endPosition = 0;

197  void OnVisitIdentifierNode(IdentifierNode &node) override {
198  const auto &objectsContainersList =
199  projectScopedContainers.GetObjectsContainersList();
200  auto type = gd::ExpressionTypeFinder::GetType(
201  platform, projectScopedContainers, rootType, node);
202  if (gd::ParameterMetadata::IsObject(type)) {
203  AddColoration(gd::ExpressionColorationDescription::ColorationKind::Object,
204  node.identifierNameLocation.GetStartPosition(),
205  node.identifierNameLocation.GetEndPosition());
206 
207  } else if (gd::ValueTypeMetadata::IsTypeLegacyPreScopedVariable(type)) {
208  AddColoration(
209  gd::ExpressionColorationDescription::ColorationKind::Variable,
210  node.location.GetStartPosition(), node.location.GetStartPosition());
211 
212  } else {
213  // Might be:
214  // - An object variable, object behavior or object expression.
215  // - Or a variable with a child.
216  projectScopedContainers.MatchIdentifierWithName<void>(
217  node.identifierName,
218  [&]() {
219  // This is an object.
220  AddColoration(

222  node.identifierNameLocation.GetStartPosition(),
223  node.identifierNameLocation.GetEndPosition());
224 
225  if (node.childIdentifierNameLocation.IsValid()) {
226  AddColoration(
227  gd::ExpressionColorationDescription::ColorationKind::Variable,
228  node.childIdentifierNameLocation.GetStartPosition(),
229  node.childIdentifierNameLocation.GetEndPosition());
230  }
231  },
232  [&]() {
233  // This is a variable.
234  AddColoration(
235  gd::ExpressionColorationDescription::ColorationKind::Variable,
236  node.location.GetStartPosition(),
237  node.location.GetEndPosition());
238  },
239  [&]() {
240  // This is a property.
241  AddColoration(
242  gd::ExpressionColorationDescription::ColorationKind::Variable,
243  node.location.GetStartPosition(),
244  node.location.GetEndPosition());
245  },
246  [&]() {
247  // This is a parameter.
248  AddColoration(
249  gd::ExpressionColorationDescription::ColorationKind::Variable,
250  node.location.GetStartPosition(),
251  node.location.GetEndPosition());
252  },
253  [&]() {
254  // Ignore unrecognised identifiers here.
255  });
256  }

261  void OnVisitFunctionCallNode(FunctionCallNode &node) override {
262  auto type = gd::ExpressionTypeFinder::GetType(
263  platform, projectScopedContainers, rootType, node);
264  auto functionColorationKind =
265  gd::ParameterMetadata::IsExpression("number", type)
266  ? gd::ExpressionColorationDescription::ColorationKind::Number
267  : gd::ExpressionColorationDescription::ColorationKind::String;
268 
269  if (node.objectNameLocation.IsValid()) {
270  AddColoration(gd::ExpressionColorationDescription::ColorationKind::Object,
271  node.objectNameLocation.GetStartPosition(),
272  node.objectNameLocation.GetEndPosition());
273  AddColoration(functionColorationKind,
274  node.objectNameLocation.GetEndPosition(),
275  node.openingParenthesisLocation.GetEndPosition());
276  } else {
277  AddColoration(functionColorationKind, node.location.GetStartPosition(),
278  node.openingParenthesisLocation.GetEndPosition());
279  }
280  auto previousParameterEndPosition =
281  node.openingParenthesisLocation.GetEndPosition();
282  for (auto &&parameter : node.parameters) {
283  if (previousParameterEndPosition <
284  parameter->location.GetStartPosition()) {
285  // Handle commas
286  AddColoration(functionColorationKind, previousParameterEndPosition,
287  parameter->location.GetStartPosition());
288  }
289  parameter->Visit(*this);
290  previousParameterEndPosition = parameter->location.GetEndPosition();
291  }
292  AddColoration(functionColorationKind,
293  node.closingParenthesisLocation.GetStartPosition(),
294  node.location.GetEndPosition());
295  }
296  void OnVisitEmptyNode(EmptyNode &node) override {}

306  void AddColoration(
307  gd::ExpressionColorationDescription::ColorationKind colorationKind,
308  size_t startPosition, size_t endPosition) {
309  if (!colorations.empty() &&
310  colorations.back().GetColorationKind() == colorationKind &&
311  colorations.back().GetEndPosition() == startPosition) {
312  colorations.back().SetEndPosition(endPosition);
313  } else {
314  gd::ExpressionColorationDescription coloration(
315  colorationKind, startPosition, endPosition);
316  colorations.push_back(coloration);
317  }
318  }
