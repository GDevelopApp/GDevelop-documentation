141  template <class ReturnType>
142  ReturnType MatchIdentifierWithName(
143  const gd::String &name,
144  std::function<ReturnType()> objectCallback,
145  std::function<ReturnType()> variableCallback,
146  std::function<ReturnType()> propertyCallback,
147  std::function<ReturnType()> parameterCallback,
148  std::function<ReturnType()> notFoundCallback) const {
149  if (objectsContainersList.HasObjectOrGroupNamed(name))
150  return objectCallback();
151  else if (variablesContainersList.Has(name))
152  return variableCallback();
153  else if (ParameterMetadataTools::Has(parametersVectorsList, name))
154  return parameterCallback();
155  else if (propertiesContainersList.Has(name))
156  return propertyCallback();
157 
158  return notFoundCallback();
159  };

174  objectsContainersList.ForEachNameMatchingSearch(

178  if (namesAlreadySeen.count(name) == 0) {
179  namesAlreadySeen.insert(name);
180  objectCallback(name, objectConfiguration);
181  }
182  });
183  variablesContainersList.ForEachVariableMatchingSearch(
184  search, [&](const gd::String &name, const gd::Variable &variable) {
185  if (namesAlreadySeen.count(name) == 0) {
186  namesAlreadySeen.insert(name);
187  variableCallback(name, variable);
188  }
189  });
190  gd::ParameterMetadataTools::ForEachParameterMatchingSearch(
191  parametersVectorsList,
192  search,
193  [&](const gd::ParameterMetadata &parameter) {
194  if (namesAlreadySeen.count(parameter.GetName()) == 0) {
195  namesAlreadySeen.insert(parameter.GetName());
196  parameterCallback(parameter);
197  }
198  });
199  propertiesContainersList.ForEachPropertyMatchingSearch(
200  search, [&](const gd::NamedPropertyDescriptor &property) {
201  if (namesAlreadySeen.count(property.GetName()) == 0) {
202  namesAlreadySeen.insert(property.GetName());
203  propertyCallback(property);
204  }
205  });
