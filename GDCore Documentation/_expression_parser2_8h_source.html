90  std::unique_ptr<ExpressionNode> Start() {
91  size_t expressionStartPosition = GetCurrentPosition();
92  auto expression = Expression();
93 
94  // Check for extra characters at the end of the expression
95  if (!IsEndReached()) {
96  auto op = gd::make_unique<OperatorNode>(' ');

98  op->rightHandSide = ReadUntilEnd();

103  "removed (or completed if your expression is not finished)."));

113  std::unique_ptr<ExpressionNode> Expression() {
114  SkipAllWhitespaces();
115 
116  size_t expressionStartPosition = GetCurrentPosition();
117  std::unique_ptr<ExpressionNode> leftHandSide = Term();
118 
119  SkipAllWhitespaces();
120 
121  if (IsEndReached()) return leftHandSide;
122  if (CheckIfChar(IsExpressionEndingChar)) return leftHandSide;
123  if (CheckIfChar(IsExpressionOperator)) {
124  auto op = gd::make_unique<OperatorNode>(GetCurrentChar());
125  op->leftHandSide = std::move(leftHandSide);
126  op->leftHandSide->parent = op.get();
127  op->diagnostic = ValidateOperator(GetCurrentChar());
128  SkipChar();
129  op->rightHandSide = Expression();
130  op->rightHandSide->parent = op.get();
131 
132  op->location = ExpressionParserLocation(expressionStartPosition,
133  GetCurrentPosition());
134  return std::move(op);
135  }
136 
137  leftHandSide->diagnostic = RaiseSyntaxError(
138  "More than one term was found. Verify that your expression is "
139  "properly written.");
140 
141  auto op = gd::make_unique<OperatorNode>(' ');
142  op->leftHandSide = std::move(leftHandSide);
143  op->leftHandSide->parent = op.get();
144  op->rightHandSide = Expression();
145  op->rightHandSide->parent = op.get();
146  op->location =
147  ExpressionParserLocation(expressionStartPosition, GetCurrentPosition());
148  return std::move(op);

151  std::unique_ptr<ExpressionNode> Term() {
152  SkipAllWhitespaces();
153 
154  size_t expressionStartPosition = GetCurrentPosition();
155  std::unique_ptr<ExpressionNode> factor = Factor();
156 
157  SkipAllWhitespaces();
158 
159  // This while loop is used instead of a recursion (like in Expression)
160  // to guarantee the proper operator precedence. (Expression could also
161  // be reworked to use a while loop).
162  while (CheckIfChar(IsTermOperator)) {
163  auto op = gd::make_unique<OperatorNode>(GetCurrentChar());
164  op->leftHandSide = std::move(factor);
165  op->leftHandSide->parent = op.get();
166  op->diagnostic = ValidateOperator(GetCurrentChar());
167  SkipChar();
168  op->rightHandSide = Factor();
169  op->rightHandSide->parent = op.get();
170  op->location = ExpressionParserLocation(expressionStartPosition,
171  GetCurrentPosition());
172  SkipAllWhitespaces();
173 
174  factor = std::move(op);
175  }
176 
177  return factor;
178  };

180  std::unique_ptr<ExpressionNode> Factor() {
181  SkipAllWhitespaces();
182  size_t expressionStartPosition = GetCurrentPosition();
183 
184  if (CheckIfChar(IsQuote)) {
185  std::unique_ptr<ExpressionNode> factor = ReadText();
186  return factor;
187  } else if (CheckIfChar(IsUnaryOperator)) {
188  auto unaryOperatorCharacter = GetCurrentChar();
189  SkipChar();
190 
191  auto operatorOperand = Factor();
192 
193  auto unaryOperator = gd::make_unique<UnaryOperatorNode>(
194  unaryOperatorCharacter);
195  unaryOperator->diagnostic = ValidateUnaryOperator(
196  unaryOperatorCharacter, expressionStartPosition);
197  unaryOperator->factor = std::move(operatorOperand);
198  unaryOperator->factor->parent = unaryOperator.get();
199  unaryOperator->location = ExpressionParserLocation(
200  expressionStartPosition, GetCurrentPosition());
201 
202  return std::move(unaryOperator);
203  } else if (CheckIfChar(IsNumberFirstChar)) {
204  std::unique_ptr<ExpressionNode> factor = ReadNumber();
205  return factor;
206  } else if (CheckIfChar(IsOpeningParenthesis)) {
207  SkipChar();
208  std::unique_ptr<ExpressionNode> factor = SubExpression();
209 
210  if (!CheckIfChar(IsClosingParenthesis)) {
211  factor->diagnostic =
212  RaiseSyntaxError(_("Missing a closing parenthesis. Add a closing "
213  "parenthesis for each opening parenthesis."));
214  }
215  SkipIfChar(IsClosingParenthesis);
216  return factor;
217  } else if (CheckIfChar(IsAllowedInIdentifier)) {
218  return Identifier();
219  }
220 
221  std::unique_ptr<ExpressionNode> factor = ReadUntilWhitespace();
222  return factor;
223  }

238  std::unique_ptr<IdentifierOrFunctionCallOrObjectFunctionNameOrEmptyNode>
239  Identifier() {
240  auto identifierAndLocation = ReadIdentifierName();
241  gd::String name = identifierAndLocation.name;
242  auto nameLocation = identifierAndLocation.location;
243 
244  SkipAllWhitespaces();
245 
246  // We consider a namespace separator to be allowed here and be part of the
247  // function name (or object name, but object names are not allowed to
248  // contain a ":"). This is because functions from extensions have their
249  // extension name prefix, and separated by the namespace separator. This
250  // could maybe be refactored to create different nodes in the future.
251  if (IsNamespaceSeparator()) {
252  SkipNamespaceSeparator();
253  SkipAllWhitespaces();
254 
255  auto postNamespaceIdentifierAndLocation = ReadIdentifierName();
256  name += NAMESPACE_SEPARATOR;
257  name += postNamespaceIdentifierAndLocation.name;
258  ExpressionParserLocation completeNameLocation(
259  nameLocation.GetStartPosition(),
260  postNamespaceIdentifierAndLocation.location.GetEndPosition());
261  nameLocation = completeNameLocation;
262  }
263 
264  if (CheckIfChar(IsOpeningParenthesis)) {
265  ExpressionParserLocation openingParenthesisLocation = SkipChar();
266  return FreeFunction(name, nameLocation, openingParenthesisLocation);
267  } else if (CheckIfChar(IsDot)) {
268  ExpressionParserLocation dotLocation = SkipChar();
269  SkipAllWhitespaces();
270  return ObjectFunctionOrBehaviorFunction(
271  name, nameLocation, dotLocation);
272  } else if (CheckIfChar(IsOpeningSquareBracket)) {
273  return Variable(name, nameLocation);
274  }
275  else {
276  auto identifier = gd::make_unique<IdentifierNode>(name);
277  identifier->location = ExpressionParserLocation(
278  nameLocation.GetStartPosition(), GetCurrentPosition());
279  identifier->identifierNameLocation = identifier->location;
280  return std::move(identifier);
281  }
282  }

361  ObjectFunctionOrBehaviorFunction(
362  const gd::String &parentIdentifier,
363  const ExpressionParserLocation &parentIdentifierLocation,

377  childIdentifierName,
378  parentIdentifierLocation,
379  parentIdentifierDotLocation,

403  auto child =
404  gd::make_unique<VariableAccessorNode>(childIdentifierName);
405  child->child = VariableAccessorOrVariableBracketAccessor();
406  child->child->parent = child.get();
407  child->nameLocation = childIdentifierNameLocation;
408  child->dotLocation = parentIdentifierDotLocation;
409  child->location = ExpressionParserLocation(

412  variable->child->parent = variable.get();

414  variable->location = ExpressionParserLocation(
415  parentIdentifierLocation.GetStartPosition(), GetCurrentPosition());
416  variable->nameLocation = parentIdentifierLocation;
417 
418  return std::move(variable);
419  }

423  if (!CheckIfChar(IsParameterSeparator) && !CheckIfChar(IsClosingParenthesis) && !IsEndReached()) {
424  node->diagnostic = RaiseSyntaxError(

430  parentIdentifierLocation.GetStartPosition(), GetCurrentPosition());
431  node->identifierNameLocation = parentIdentifierLocation;
432  node->identifierNameDotLocation = parentIdentifierDotLocation;
433  node->childIdentifierNameLocation = childIdentifierNameLocation;
434  return std::move(node);

492  // A temporary node that will be integrated into function nodes.
493  struct ParametersNode {
494  std::vector<std::unique_ptr<ExpressionNode>> parameters;
495  std::unique_ptr<gd::ExpressionParserError> diagnostic;

499  ParametersNode Parameters(
500  FunctionCallNode *functionCallNode,
501  const gd::String &objectName = "",
502  const gd::String &behaviorName = "") {
503  std::vector<std::unique_ptr<ExpressionNode>> parameters;
504  gd::String lastObjectName = "";
505 
506  // By convention, object is always the first parameter, and behavior the
507  // second one.
508  size_t parameterIndex =
509  WrittenParametersFirstIndex(objectName, behaviorName);
510 
511  bool previousCharacterIsParameterSeparator = false;
512  while (!IsEndReached()) {
513  SkipAllWhitespaces();
514 
515  if (CheckIfChar(IsClosingParenthesis) && !previousCharacterIsParameterSeparator) {
516  auto closingParenthesisLocation = SkipChar();
517  return ParametersNode{
518  std::move(parameters), nullptr, closingParenthesisLocation};
519  }
520  bool isEmptyParameter = CheckIfChar(IsParameterSeparator)
521  || (CheckIfChar(IsClosingParenthesis) && previousCharacterIsParameterSeparator);
522  auto parameter = isEmptyParameter ? gd::make_unique<EmptyNode>() : Expression();
523  parameter->parent = functionCallNode;
524  parameters.push_back(std::move(parameter));
525 
526  SkipAllWhitespaces();
527  previousCharacterIsParameterSeparator = CheckIfChar(IsParameterSeparator);
528  SkipIfChar(IsParameterSeparator);
529  parameterIndex++;
530  }
531 
532  ExpressionParserLocation invalidClosingParenthesisLocation;
533  return ParametersNode{
534  std::move(parameters),
535  RaiseSyntaxError(_("The list of parameters is not terminated. Add a "
536  "closing parenthesis to end the parameters.")),
537  invalidClosingParenthesisLocation};
538  }

593  ExpressionParserLocation SkipNamespaceSeparator() {
594  size_t startPosition = currentPosition;
595  // Namespace separator is a special kind of delimiter as it is 2 characters
596  // long
597  if (IsNamespaceSeparator()) {
598  currentPosition += NAMESPACE_SEPARATOR.size();
599  }
600 
601  return ExpressionParserLocation(startPosition, currentPosition);
602  }
603 
604  bool CheckIfChar(
605  const std::function<bool(gd::String::value_type)> &predicate) {
606  if (currentPosition >= expression.size()) return false;
607  gd::String::value_type character = expression[currentPosition];
608 
609  return predicate(character);
610  }
611 
612  bool IsNamespaceSeparator() {
613  // Namespace separator is a special kind of delimiter as it is 2 characters
614  // long
615  return (currentPosition + NAMESPACE_SEPARATOR.size() <= expression.size() &&
616  expression.substr(currentPosition, NAMESPACE_SEPARATOR.size()) ==
617  NAMESPACE_SEPARATOR);
618  }

620  bool IsEndReached() { return currentPosition >= expression.size(); }

622  // A temporary node used when reading an identifier
623  struct IdentifierAndLocation {
624  gd::String name;
625  ExpressionParserLocation location;

627 
628  IdentifierAndLocation ReadIdentifierName() {
629  gd::String name;
630  size_t startPosition = currentPosition;
631  while (currentPosition < expression.size() &&
632  (CheckIfChar(IsAllowedInIdentifier)
633  // Allow whitespace in identifier name for compatibility
634  || expression[currentPosition] == ' ')) {
635  name += expression[currentPosition];
636  currentPosition++;
637  }
638 
639  // Trim whitespace at the end (we allow them for compatibility inside
640  // the name, but after the last character that is not whitespace, they
641  // should be ignore again).
642  if (!name.empty() && IsWhitespace(name[name.size() - 1])) {
643  size_t lastCharacterPos = name.size() - 1;
644  while (lastCharacterPos < name.size() &&
645  IsWhitespace(name[lastCharacterPos])) {
646  lastCharacterPos--;
647  }
648  if ((lastCharacterPos + 1) < name.size()) {
649  name.erase(lastCharacterPos + 1);
650  }
651  }
652 
653  IdentifierAndLocation identifierAndLocation{
654  name,
655  // The location is ignoring the trailing whitespace (only whitespace
656  // inside the identifier are allowed for compatibility).
657  ExpressionParserLocation(startPosition, startPosition + name.size())};
658  return identifierAndLocation;
659  }
