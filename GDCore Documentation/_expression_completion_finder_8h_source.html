350  void OnVisitTextNode(TextNode& node) override {
351  // Completions are searched in the case the text node is a parameter of a
352  // function call.
353  FunctionCallNode* functionCall =
354  dynamic_cast<FunctionCallNode*>(maybeParentNodeAtLocation);
355  if (functionCall != nullptr) {
356 
357  int parameterIndex = -1;
358  for (int i = 0; i < functionCall->parameters.size(); i++) {
359  if (functionCall->parameters.at(i).get() == &node) {
360  parameterIndex = i;
361  break;
362  }
363  }
364  if (parameterIndex < 0) {
365  return;
366  }
367  // Search the parameter metadata index skipping invisible ones.
368  size_t visibleParameterIndex = 0;
369  size_t metadataParameterIndex =
370  ExpressionParser2::WrittenParametersFirstIndex(
371  functionCall->objectName, functionCall->behaviorName);
372  const gd::ExpressionMetadata &metadata = MetadataProvider::GetFunctionCallMetadata(
373  platform, globalObjectsContainer, objectsContainer, *functionCall);
374 
375  const gd::ParameterMetadata* parameterMetadata = nullptr;
376  while (metadataParameterIndex <
377  metadata.parameters.size()) {
378  if (!metadata.parameters[metadataParameterIndex]
379  .IsCodeOnly()) {
380  if (visibleParameterIndex == parameterIndex) {
381  parameterMetadata = &metadata.parameters[metadataParameterIndex];
382  }
383  visibleParameterIndex++;
384  }
385  metadataParameterIndex++;
386  }
387  const int visibleParameterCount = visibleParameterIndex;
388  if (parameterMetadata == nullptr) {
389  // There are too many parameters in the expression, this text node is
390  // not actually linked to a parameter expected by the function call.
391  return;
392  }
393 
394  const gd::String& type = parameterMetadata->GetType();
395  if (type == "string") {
396  // No completions for an arbitrary string.
397  return;
398  }
399 
400  bool isLastParameter = parameterIndex == visibleParameterCount - 1;
401  completions.push_back(ExpressionCompletionDescription::ForText(
402  type,
403  *parameterMetadata,
404  node.text,
405  node.location.GetStartPosition(),
406  node.location.GetEndPosition(),
407  isLastParameter,
408  functionCall->objectName));
409  }
410  }
411  void OnVisitVariableNode(VariableNode& node) override {
412  auto type = gd::ExpressionTypeFinder::GetType(platform, globalObjectsContainer, objectsContainer, rootType, node);
413  auto objectName = gd::ExpressionVariableOwnerFinder::GetObjectName(
414  platform,
415  globalObjectsContainer,
416  objectsContainer,
417  // Variable fields doesn't use expression completion,
418  // so the object will be found inside the expression itself.
419  "",

435  void OnVisitIdentifierNode(IdentifierNode& node) override {
436  auto type = gd::ExpressionTypeFinder::GetType(platform, globalObjectsContainer, objectsContainer, rootType, node);
437  if (gd::ParameterMetadata::IsObject(type)) {
438  // Only show completions of objects if an object is required
439  completions.push_back(ExpressionCompletionDescription::ForObject(
440  type,
441  node.identifierName,
442  node.location.GetStartPosition(),
443  node.location.GetEndPosition()));
444  } else if (gd::ParameterMetadata::IsExpression("variable", type)) {
445  auto objectName = gd::ExpressionVariableOwnerFinder::GetObjectName(
446  platform,
447  globalObjectsContainer,
448  objectsContainer,
449  // Variable fields doesn't use expression completion,
450  // so the object will be found inside the expression itself.
451  "",
452  node);
453  completions.push_back(ExpressionCompletionDescription::ForVariable(
454  type,
455  node.identifierName,
456  node.location.GetStartPosition(),
457  node.location.GetEndPosition(),
458  objectName));
459  } else {
460  // Object function or behavior name
461  if (IsCaretOn(node.identifierNameLocation)) {
462  completions.push_back(ExpressionCompletionDescription::ForObject(
463  type,
464  node.identifierName,
465  node.identifierNameLocation.GetStartPosition(),
466  node.identifierNameLocation.GetEndPosition()));
467  if (!node.identifierNameDotLocation.IsValid()) {
468  completions.push_back(ExpressionCompletionDescription::ForExpression(
469  type,
470  node.identifierName,
471  node.identifierNameLocation.GetStartPosition(),
472  node.identifierNameLocation.GetEndPosition()));
473  }
474  } else if (IsCaretOn(node.identifierNameDotLocation) ||
475  IsCaretOn(node.childIdentifierNameLocation)) {
476  completions.push_back(ExpressionCompletionDescription::ForBehavior(
477  node.childIdentifierName,
478  node.childIdentifierNameLocation.GetStartPosition(),
479  node.childIdentifierNameLocation.GetEndPosition(),
480  node.identifierName));
481  completions.push_back(ExpressionCompletionDescription::ForExpression(
482  type,
483  node.childIdentifierName,
484  node.childIdentifierNameLocation.GetStartPosition(),
485  node.childIdentifierNameLocation.GetEndPosition(),
486  node.identifierName));
487  }
488  }
489  }
490  void OnVisitObjectFunctionNameNode(ObjectFunctionNameNode& node) override {
491  auto type = gd::ExpressionTypeFinder::GetType(platform, globalObjectsContainer, objectsContainer, rootType, node);
492  if (!node.behaviorFunctionName.empty() ||
493  node.behaviorNameNamespaceSeparatorLocation.IsValid()) {
494  // Behavior function (or behavior function being written, with the
495  // function name missing)
496  if (IsCaretOn(node.objectNameLocation)) {
497  completions.push_back(ExpressionCompletionDescription::ForObject(
498  type,
499  node.objectName,
500  node.objectNameLocation.GetStartPosition(),
501  node.objectNameLocation.GetEndPosition()));
502  } else if (IsCaretOn(node.objectNameDotLocation) ||
503  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
504  completions.push_back(ExpressionCompletionDescription::ForBehavior(
505  node.objectFunctionOrBehaviorName,
506  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
507  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
508  node.objectName));
509  } else if (IsCaretOn(node.behaviorNameNamespaceSeparatorLocation) ||
510  IsCaretOn(node.behaviorFunctionNameLocation)) {
511  completions.push_back(ExpressionCompletionDescription::ForExpression(
512  type,
513  node.behaviorFunctionName,
514  node.behaviorFunctionNameLocation.GetStartPosition(),
515  node.behaviorFunctionNameLocation.GetEndPosition(),
516  node.objectName,
517  node.objectFunctionOrBehaviorName));
518  }
519  } else {
520  // Object function or behavior name
521  if (IsCaretOn(node.objectNameLocation)) {
522  completions.push_back(ExpressionCompletionDescription::ForObject(
523  type,
524  node.objectName,
525  node.objectNameLocation.GetStartPosition(),
526  node.objectNameLocation.GetEndPosition()));
527  } else if (IsCaretOn(node.objectNameDotLocation) ||
528  IsCaretOn(node.objectFunctionOrBehaviorNameLocation)) {
529  completions.push_back(ExpressionCompletionDescription::ForBehavior(
530  node.objectFunctionOrBehaviorName,
531  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
532  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
533  node.objectName));
534  completions.push_back(ExpressionCompletionDescription::ForExpression(
535  type,
536  node.objectFunctionOrBehaviorName,
537  node.objectFunctionOrBehaviorNameLocation.GetStartPosition(),
538  node.objectFunctionOrBehaviorNameLocation.GetEndPosition(),
539  node.objectName));
540  }
541  }

543  void OnVisitFunctionCallNode(FunctionCallNode& node) override {
544  auto type = gd::ExpressionTypeFinder::GetType(platform, globalObjectsContainer, objectsContainer, rootType, node);
545  bool isCaretOnParenthesis = IsCaretOn(node.openingParenthesisLocation) ||
546  IsCaretOn(node.closingParenthesisLocation);
547 
548  if (!node.behaviorName.empty()) {
549  // Behavior function
550  if (IsCaretOn(node.objectNameLocation)) {
551  completions.push_back(ExpressionCompletionDescription::ForObject(
552  type,
553  node.objectName,
554  node.objectNameLocation.GetStartPosition(),
555  node.objectNameLocation.GetEndPosition()));
556  } else if (IsCaretOn(node.objectNameDotLocation) ||
557  IsCaretOn(node.behaviorNameLocation)) {
558  completions.push_back(ExpressionCompletionDescription::ForBehavior(
559  node.behaviorName,
560  node.behaviorNameLocation.GetStartPosition(),
561  node.behaviorNameLocation.GetEndPosition(),
562  node.objectName));
563  } else {
564  completions.push_back(ExpressionCompletionDescription::ForExpression(
565  type,
566  node.functionName,
567  node.functionNameLocation.GetStartPosition(),
568  node.functionNameLocation.GetEndPosition(),
569  node.objectName,
570  node.behaviorName)
571  .SetIsExact(isCaretOnParenthesis));
572  }
573  } else if (!node.objectName.empty()) {
574  // Object function
575  if (IsCaretOn(node.objectNameLocation)) {
576  completions.push_back(ExpressionCompletionDescription::ForObject(
577  type,
578  node.objectName,
579  node.objectNameLocation.GetStartPosition(),
580  node.objectNameLocation.GetEndPosition()));
581  } else {
582  // Add completions for behaviors, because we could imagine that the user
583  // wants to move from an object function to a behavior function, and so
584  // need behavior completions. Do this unless we're on the parenthesis
585  // (at which point we're only showing informative message about the
586  // function).
587  if (!isCaretOnParenthesis) {
588  completions.push_back(ExpressionCompletionDescription::ForBehavior(
589  node.functionName,
590  node.objectNameLocation.GetStartPosition(),
591  node.objectNameLocation.GetEndPosition(),
592  node.objectName));
593  }
594 
595  completions.push_back(ExpressionCompletionDescription::ForExpression(
596  type,
597  node.functionName,
598  node.functionNameLocation.GetStartPosition(),
599  node.functionNameLocation.GetEndPosition(),
600  node.objectName)
601  .SetIsExact(isCaretOnParenthesis));
602  }
603  } else {
604  // Free function
605  completions.push_back(ExpressionCompletionDescription::ForExpression(
606  type,
607  node.functionName,
608  node.functionNameLocation.GetStartPosition(),
609  node.functionNameLocation.GetEndPosition())
610  .SetIsExact(isCaretOnParenthesis));
611  }
612  }

615  completions.push_back(ExpressionCompletionDescription::ForObject(
616  type,
617  node.text,
618  node.location.GetStartPosition(),
619  node.location.GetEndPosition()));
620  completions.push_back(ExpressionCompletionDescription::ForExpression(
621  type,
622  node.text,
623  node.location.GetStartPosition(),
624  node.location.GetEndPosition()));
625  }

628  bool IsCaretOn(const ExpressionParserLocation& location,
629  bool inclusive = false) {
630  if (!location.IsValid()) return false;
631 
632  return (location.GetStartPosition() <= searchedPosition &&
633  ((!inclusive && searchedPosition < location.GetEndPosition()) ||
634  (inclusive && searchedPosition <= location.GetEndPosition())));
635  }
