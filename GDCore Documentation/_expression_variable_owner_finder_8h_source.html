113  void OnVisitFunctionCallNode(FunctionCallNode& functionCall) override {
114  if (variableNode == nullptr) {
115  return;
116  }
117  int parameterIndex = -1;
118  for (int i = 0; i < functionCall.parameters.size(); i++) {
119  if (functionCall.parameters.at(i).get() == variableNode) {
120  parameterIndex = i;
121  break;
122  }
123  }
124  if (parameterIndex < 0) {
125  return;
126  }
127  const gd::ParameterMetadata* parameterMetadata =
128  MetadataProvider::GetFunctionCallParameterMetadata(
129  platform,
130  globalObjectsContainer,
131  objectsContainer,
132  functionCall,
133  parameterIndex);
134  if (parameterMetadata == nullptr
135  || parameterMetadata->GetType() != "objectvar") {
136  return;
137  }
138 
139  // The object on which the function is called is returned if no previous
140  // parameters are objects.
141  objectName = functionCall.objectName;
142  for (int previousIndex = parameterIndex - 1; previousIndex >= 0; previousIndex--) {
143  const gd::ParameterMetadata* previousParameterMetadata =
144  MetadataProvider::GetFunctionCallParameterMetadata(
145  platform,
146  globalObjectsContainer,
147  objectsContainer,
148  functionCall,
149  previousIndex);
150  if (previousParameterMetadata != nullptr
151  && gd::ParameterMetadata::IsObject(previousParameterMetadata->GetType())) {
152  auto previousParameterNode = functionCall.parameters[previousIndex].get();
153  IdentifierNode* objectNode = dynamic_cast<IdentifierNode*>(previousParameterNode);
154  objectName = objectNode->identifierName;
155  return;
156  }
157  }
158  }
